Data Description

Column Description
id The unique ID assigned to every loan application.
Loan Amount The loan amount applied for.(in $$)
Term The number of months to repay the loan. (in months)
State The state the applicant is applying from.
Annual Income The applicant's stated annual income. (in $$)
Income Verification Status Has the bank verified the applicant's income information? Values are Verified, Partially Verified and Not Verified.
Average Account Balance The applicant's average balance of all cash accounts over a period, e.g. bank and brokerage. (in $$)
Due Amount Total late fees due from the applicant. (in $$).
Home Ownership The home-ownership status provided by the applicant. Values are: RENT, OWN, MORTGAGE, OTHER.
Loan Purpose The purpose of the loan.
Due Settlement Has the applicant has opted for settlement against any loan in the past? Values are N (not opted) and Y (opted).
Installment Amount The monthly payment owed by the borrower if the loan gets approved. (in $$)
Payment Plan Indicates if the borrower has suggested a loan repayment plan or not. Values are n (not suggested) and y (suggested).
Approve Loan The approval status of the loan. Values are 0 and 1 where 0 indicates loan is not approved while 1 indicates the loan is approved.

class basic_preprocessing:
    
    def __init__(self, X, y, numerical_cols, categorical_cols):
        self.X = X
        self.y = y
        self.numerical_cols = numerical_cols
        self.categorical_cols = categorical_cols
        self.dictionary = {}
        
    def imputation(self):
        # here only the numerical columns will be captured and null values will be replaced with median
        self.X = self.X.fillna(self.X.median())
        
        #running a loop to convert null from categorical columns with mode
        for i in self.categorical_cols:
            self.X[i] = self.X[i].fillna(self.X[i].mode().iloc[0])
            
    def outlier_treatment(self):
        for i in self.numerical_cols:
            percentile_25 = self.X[i].quantile(0.25)
            percentile_75 = self.X[i].quantile(0.75)
            iqr = percentile_75 - percentile_25
            upper_limit = percentile_75 + 1.5 * iqr
            lower_limit = percentile_25 - 1.5 * iqr
            # print('column: ', i, 'upper limit: ', upper_limit, 'lower_limit: ', lower_limit)
            
            self.X = self.X[(self.X[i] < upper_limit) & (self.X[i] > lower_limit)]
            
            
            # to see if we can afford removing all rows with any outlier value
            self.dictionary[i] = [upper_limit, lower_limit]
        print(self.dictionary)
        
        
    def calculate_outlier_percentage(self, row, threshold=2):
        z_scores = np.abs((row - row.mean()) / row.std())
        outliers_percentage = (z_scores > threshold).sum() / len(row) * 100
        return outliers_percentage
        
            
    
    def dummification(self):
        # not putting state column here as already target encoding is done for state column
        self.X = pd.get_dummies(self.X, columns = [x for x in self.categorical_cols if x not in 'State'], drop_first = True)
    
    
    def preprocessing(self):
        self.imputation()
        print('before outlier treatment: ', self.X.shape)
        # self.outlier_treatment()
        # self.X['Outlier_Percentage'] = self.X.apply(self.calculate_outlier_percentage, axis = 1)
        print('after outlier treatment: ', self.X.shape)
        print(self.X.columns)
        print(self.categorical_cols)
        self.dummification()
        
        return self.X, self.y, self.dictionary

def calc_vif(X):

    # Calculating VIF
    vif = pd.DataFrame()
    vif["variables"] = X.columns
    vif["VIF"] = [variance_inflation_factor(X.values, i) for i in range(X.shape[1])]

    return(vif)
